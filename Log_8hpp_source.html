74 // Inspired by techniques developed by Rodrigo Fernandes.
75 class Log {
76  template <typename ValueT>
77  friend Log& operator<<(Log& log, ValueT&& val)
78  {
79  log.os_ << std::forward<ValueT>(val);
80  return log;
81  }
82 
83  public:
84  enum : int {
86  Crit,
88  Error,
90  Warning,
92  Notice,
94  Info,
96  Debug
97  };
98 
99  explicit Log(int level) noexcept
100  : level_{level}
101  , os_{log_stream()}
102  {
103  os_.set_storage(os_.make_storage());
104  }
105  ~Log()
106  {
107  const auto size = os_.size();
108  write_log(level_, os_.release_storage(), size);
109  }
110 
111  // Copy.
112  Log(const Log&) = delete;
113  Log& operator=(const Log&) = delete;
114 
115  // Move.
116  Log(Log&&) = delete;
117  Log& operator=(Log&&) = delete;
118 
119  constexpr explicit operator bool() const { return true; }
121  Log& operator()(const char* data, std::streamsize size)
122  {
123  os_.write(data, size);
124  return *this;
125  }
127  Log& operator()() noexcept { return *this; }
128 
129  private:
130  const int level_;
131  LogStream& os_;
132 };
