104 void std_logger(int level, LogMsgPtr msg, std::size_t size) noexcept
105 {
106  const auto now = WallClock::now();
107  const auto t = WallClock::to_time_t(now);
108  const auto ms = ms_since_epoch(now);
109 
110  struct tm tm; // NOLINT(hicpp-member-init)
111  localtime_r(&t, &tm);
112 
113  // The following format has an upper-bound of 42 characters:
114  // "%b %d %H:%M:%S.%03d %-7s [%d]: "
115  //
116  // Example:
117  // Mar 14 00:00:00.000 WARNING [0123456789]: msg...
118  // <---------------------------------------->
119  char head[42 + 1];
120  size_t hlen = strftime(head, sizeof(head), "%b %d %H:%M:%S", &tm);
121  hlen += sprintf(head + hlen, ".%03d %-7s [%d]: ", static_cast<int>(ms % 1000), log_label(level),
122  static_cast<int>(gettid()));
123  char tail{'\n'};
124  iovec iov[] = {
125  {head, hlen}, //
126  {msg.get(), size}, //
127  {&tail, 1} //
128  };
129 
130  int fd{level > Log::Error ? STDOUT_FILENO : STDERR_FILENO};
131  // The following lock was required to avoid interleaving.
132  lock_guard<mutex> lock{mutex_};
133  // Best effort given that this is the logger.
134 #pragma GCC diagnostic push
135 #pragma GCC diagnostic ignored "-Wunused-result"
136  writev(fd, iov, sizeof(iov) / sizeof(iov[0]));
137 #pragma GCC diagnostic pop
138 }

140 void sys_logger(int level, LogMsgPtr msg, std::size_t size) noexcept
141 {
142  int prio;
143  switch (level) {
144  case Log::Crit:
145  prio = LOG_CRIT;
146  break;
147  case Log::Error:
148  prio = LOG_ERR;
149  break;
150  case Log::Warning:
151  prio = LOG_WARNING;
152  break;
153  case Log::Notice:
154  prio = LOG_NOTICE;
155  break;
156  case Log::Info:
157  prio = LOG_INFO;
158  break;
159  default:
160  prio = LOG_DEBUG;
161  }
162  syslog(prio, "%.*s", static_cast<int>(size), static_cast<const char*>(msg.get()));
163 }
